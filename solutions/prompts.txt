class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        self.counter = 0
        self.res = 0
        self.dfs(root)
        return self.res


    def dfs(self,node):
        if node is None:
            return
        self.dfs(node.left)
        self.counter+=1
        if(self.counter == k):
            self.res = node.val
            return
        self.dfs(node.right)
